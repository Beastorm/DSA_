### String Basic Program

**1) Reverse a string (in-place):**
``` cpp
#include <bits/stdc++.h>
using namespace std;

void reverseString(string &s) {
    int i = 0, j = (int)s.size() - 1;
    while (i < j) {
        swap(s[i], s[j]);
        i++;
        j--;
    }
}

int main() {
    string s;
    cout << "Enter string: ";
    getline(cin, s);

    reverseString(s);
    cout << "Reversed string: " << s << "\n";
    return 0;
}
```
**2) Find length of a string (manual)**  

``` cpp
#include <bits/stdc++.h>
using namespace std;

int stringLength(const string &s) {
    int len = 0;
    for (char c : s) len++;
    return len;
}

int main() {
    string s;
    cout << "Enter string: ";
    getline(cin, s);

    cout << "Length: " << stringLength(s) << "\n";
    return 0;
}
```


**3) Check palindrome**

``` cpp
#include <bits/stdc++.h>
using namespace std;

bool isPalindrome(const string &s) {
    int i = 0, j = (int)s.size() - 1;
    while (i < j) {
        if (s[i] != s[j]) return false;
        i++;
        j--;
    }
    return true;
}

int main() {
    string s;
    cout << "Enter string: ";
    getline(cin, s);

    cout << (isPalindrome(s) ? "Palindrome" : "Not Palindrome") << "\n";
    return 0;
}
```
