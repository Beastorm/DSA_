# Kruskal's Algorithm - Minimum Spanning Tree

## Problem Statement

Given a weighted undirected connected graph with **V** vertices and **E** edges, find the **Minimum Spanning Tree (MST)** using Kruskal's algorithm. An MST is a subset of edges that:
- Connects all vertices
- Contains exactly V-1 edges
- Has a minimum total weight
- Forms no cycles

---

## Solution Overview

**Kruskal's Algorithm** uses a **greedy approach** with **Union-Find (Disjoint Set Union)**:

1. Sort all edges by weight
2. Use Union-Find to detect cycles
3. Add edges that don't create cycles
4. Stop when V-1 edges are added

```
Time Complexity: O(E log E + E α(V)) ≈ O(E log E)
Space Complexity: O(V + E)
```

## Core Data Structures

### Union-Find (DSU) Implementation

```cpp
class DSU {
private:
    vector<int> parent;
    vector<int> rank;
    
public:
    DSU(int n) {
        parent.resize(n);
        rank.resize(n, 0);
        for (int i = 0; i < n; i++) {
            parent[i] = i;
        }
    }
    
    // Path compression optimization
    int find(int x) {
        if (parent[x] != x) {
            parent[x] = find(parent[x]);
        }
        return parent[x];
    }
    
    // Union by rank optimization
    bool unite(int x, int y) {
        int px = find(x);
        int py = find(y);
        
        if (px == py) return false; // Already connected
        
        // Attach smaller rank to larger rank
        if (rank[px] < rank[py]) {
            parent[px] = py;
        } else if (rank[px] > rank[py]) {
            parent[py] = px;
        } else {
            parent[py] = px;
            rank[px]++;
        }
        return true;
    }
};
```

## Complete Solution

### Method 1: Returning MST Edges

```cpp
#include <bits/stdc++.h>
using namespace std;

struct Edge {
    int u, v, weight;
    bool operator<(const Edge& other) const {
        return weight < other.weight;
    }
};

class DSU {
private:
    vector<int> parent, rank;
    
public:
    DSU(int n) : parent(n), rank(n, 0) {
        iota(parent.begin(), parent.end(), 0);
    }
    
    int find(int x) {
        return parent[x] == x ? x : parent[x] = find(parent[x]);
    }
    
    bool unite(int x, int y) {
        int px = find(x), py = find(y);
        if (px == py) return false;
        
        if (rank[px] < rank[py]) swap(px, py);
        parent[py] = px;
        if (rank[px] == rank[py]) rank[px]++;
        return true;
    }
};

vector<Edge> kruskal(int n, vector<Edge>& edges) {
    sort(edges.begin(), edges.end());
    
    DSU dsu(n);
    vector<Edge> mst;
    
    for (const auto& e : edges) {
        if (dsu.unite(e.u, e.v)) {
            mst.push_back(e);
            if (mst.size() == n - 1) break;
        }
    }
    
    return mst;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n = 4, m = 5;
    vector<Edge> edges = {
        {0, 1, 4},
        {0, 2, 5},
        {0, 3, 3},
        {1, 3, 2},
        {2, 3, 1}
    };
    
    vector<Edge> mst = kruskal(n, edges);
    
    int totalWeight = 0;
    cout << "MST Edges:\n";
    for (const auto& e : mst) {
        cout << e.u << " - " << e.v << " : " << e.weight << "\n";
        totalWeight += e.weight;
    }
    cout << "Total Weight: " << totalWeight << "\n";
    
    return 0;
}
```

**Output:**
```

2 - 3 : 1
1 - 3 : 2
0 - 3 : 3
Total Weight: 6
```

### Method 2: Returning MST Weight

```cpp
int kruskalWeight(int n, vector<Edge>& edges) {
    sort(edges.begin(), edges.end());
    
    DSU dsu(n);
    int totalWeight = 0;
    int edgeCount = 0;
    
    for (const auto& e : edges) {
        if (dsu.unite(e.u, e.v)) {
            totalWeight += e.weight;
            edgeCount++;
            
            if (edgeCount == n - 1) break;
        }
    }
    
    // Check if graph is connected
    return (edgeCount == n - 1) ? totalWeight : -1;
}
```

## Step-by-Step Trace

### Example Graph

```
    4
   / \
  A---B
  |\ /|
5 | X | 2
  |/ \|
  C---D
   1  3
```

### Execution Trace

```
Initial: parent = [0,1,2,3], edges sorted by weight

Step 1: Edge (2,3,1) - C-D
  find(2)=2, find(3)=3 → Different
  unite(2,3): parent[3]=2
  MST: [(2,3,1)]
  Weight: 1

Step 2: Edge (1,3,2) - B-D
  find(1)=1, find(3)=2 → Different
  unite(1,3): parent[2]=1 (by rank)
  MST: [(2,3,1), (1,3,2)]
  Weight: 3

Step 3: Edge (0,3,3) - A-D
  find(0)=0, find(3)→2→1 → Different
  unite(0,3): parent[1]=0 (by rank)
  MST: [(2,3,1), (1,3,2), (0,3,3)]
  Weight: 6

Step 4: Edge (0,1,4) - A-B
  find(0)=0, find(1)→0 → Same!
  Skip (would create cycle)

Step 5: Edge (0,2,5) - A-C
  find(0)=0, find(2)→1→0 → Same!
  Skip (would create cycle)

Final MST Weight: 6
Edges: 3 (= V-1)
```

## Complete Problem Solution

```cpp
#include <bits/stdc++.h>
using namespace std;

struct Edge {
    int u, v, weight;
    int id;
    
    bool operator<(const Edge& other) const {
        return weight < other.weight;
    }
};

class DSU {
private:
    vector<int> parent, rank;
    
public:
    DSU(int n) : parent(n), rank(n, 0) {
        iota(parent.begin(), parent.end(), 0);
    }
    
    int find(int x) {
        return parent[x] == x ? x : parent[x] = find(parent[x]);
    }
    
    bool unite(int x, int y) {
        int px = find(x), py = find(y);
        if (px == py) return false;
        
        if (rank[px] < rank[py]) swap(px, py);
        parent[py] = px;
        if (rank[px] == rank[py]) rank[px]++;
        return true;
    }
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n, m;
    cin >> n >> m;
    
    vector<Edge> edges(m);
    for (int i = 0; i < m; i++) {
        cin >> edges[i].u >> edges[i].v >> edges[i].weight;
        edges[i].u--; edges[i].v--;
        edges[i].id = i;
    }
    
    sort(edges.begin(), edges.end());
    
    DSU dsu(n);
    vector<int> mstEdges;
    long long totalWeight = 0;
    
    for (const auto& e : edges) {
        if (dsu.unite(e.u, e.v)) {
            mstEdges.push_back(e.id);
            totalWeight += e.weight;
            
            if (mstEdges.size() == n - 1) break;
        }
    }
    
    // Check if connected
    if (mstEdges.size() != n - 1) {
        cout << "Graph is not connected\n";
        return 0;
    }
    
    cout << totalWeight << "\n";
    for (int id : mstEdges) {
        cout << id + 1 << " ";
    }
    cout << "\n";
    
    return 0;
}
```

## Edge Cases

### Case 1: Disconnected Graph
```cpp
// Graph cannot have spanning tree
// Check: if (mst.size() != n - 1) return -1;
```

### Case 2: Single Vertex
```cpp
// MST weight = 0, no edges needed
n=1, m=0 → weight=0
```

### Case 3: Negative Weights
```cpp
// Kruskal handles negative weights correctly
// MST definition only requires minimum total weight
edges = {(0,1,-5), (1,2,3)} → MST weight = -2
```

### Case 4: Multiple Edges
```cpp
// Only smallest weight between two vertices matters
(u,v,10), (u,v,5) → Only (u,v,5) considered
```

### Case 5: Self-Loops
```cpp
// Always create cycles, automatically skipped
(u,u,w) → find(u)=find(u), never added
```

## Key Optimizations

### 1. Path Compression
```cpp
int find(int x) {
    return parent[x] == x ? x : parent[x] = find(parent[x]);
}
// Flattens tree structure after each call
```

### 2. Union by Rank
```cpp
if (rank[px] < rank[py]) swap(px, py);
parent[py] = px;
if (rank[px] == rank[py]) rank[px]++;
// Keeps tree shallow
```

### 3. Early Termination
```cpp
if (mstEdges.size() == n - 1) break;
// Stop after finding V-1 edges
```
